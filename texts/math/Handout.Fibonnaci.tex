\begin{align*}
	F_n &>= 2^{n/2}\\
	F_n &= F_{n-1}+F_{n-2}\\
	&>= 2^{(n-1)/2} + 2^{(n-2)/2}\\

	& &= 2*2^{(n-2)/2}\\
